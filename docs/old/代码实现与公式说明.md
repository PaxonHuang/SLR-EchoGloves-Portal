# 项目关键公式的代码实现与说明

## 1. 传感器数据处理

### 1.1 分压电路计算

**理论公式：**
```
Vout = Vin * R2/(R1 + R2)
```

**代码实现：**
```c
// flex_sensor.c - 分压电路ADC值转换为电压
float flex_adc_to_voltage(int adc_raw) {
    const float VIN = 3.3f;           // ESP32供电电压3.3V
    const float ADC_MAX = 4095.0f;    // 12位ADC最大值
    const float R_PULLDOWN = 10000.0f; // 下拉电阻10kΩ
    
    // ADC原始值转换为电压
    float vout = (adc_raw * VIN) / ADC_MAX;
    
    // 根据分压公式反推柔性传感器阻值
    // Vout = Vin * R_pulldown / (R_flex + R_pulldown)
    // 解得: R_flex = R_pulldown * (Vin - Vout) / Vout
    float r_flex = R_PULLDOWN * (VIN - vout) / vout;
    
    ESP_LOGD(TAG, "ADC: %d, Voltage: %.3fV, R_flex: %.1fΩ", 
             adc_raw, vout, r_flex);
    
    return vout;
}

// 应变片电阻变化计算
float calculate_strain(float r_current, float r_reference) {
    const float GAUGE_FACTOR = 2.1f;  // 应变片灵敏度系数K
    
    // 应变计算公式: ε = (ΔR/R) / K
    float strain = ((r_current - r_reference) / r_reference) / GAUGE_FACTOR;
    
    ESP_LOGD(TAG, "Strain: %.6f", strain);
    return strain;
}
```

**说明：**
- 此代码实现了柔性传感器的电阻-电压转换
- 通过分压电路公式反推传感器阻值变化
- 应变量计算用于量化手指弯曲程度

### 1.2 数字滤波算法

**均值滤波公式：**
```
y[n] = (1/N) * Σ(k=0 to N-1) x[n-k]
```

**代码实现：**
```c
// flex_sensor.c - 均值滤波器实现
#define FILTER_WINDOW_SIZE 10

typedef struct {
    float buffer[FILTER_WINDOW_SIZE];
    int index;
    int count;
    float sum;
} mean_filter_t;

static mean_filter_t flex_filters[5];

float apply_mean_filter(int sensor_id, float new_value) {
    mean_filter_t* filter = &flex_filters[sensor_id];
    
    // 移除旧值
    if (filter->count == FILTER_WINDOW_SIZE) {
        filter->sum -= filter->buffer[filter->index];
    } else {
        filter->count++;
    }
    
    // 添加新值
    filter->buffer[filter->index] = new_value;
    filter->sum += new_value;
    
    // 更新索引（循环缓冲区）
    filter->index = (filter->index + 1) % FILTER_WINDOW_SIZE;
    
    // 返回均值
    float filtered_value = filter->sum / filter->count;
    
    ESP_LOGV(TAG, "Sensor %d: Raw=%.3f, Filtered=%.3f", 
             sensor_id, new_value, filtered_value);
    
    return filtered_value;
}
```

**中值滤波公式：**
```
y[n] = median{x[n-N/2], ..., x[n], ..., x[n+N/2]}
```

**代码实现：**
```c
// 中值滤波器实现
#define MEDIAN_WINDOW_SIZE 7

int compare_float(const void* a, const void* b) {
    float fa = *(const float*)a;
    float fb = *(const float*)b;
    return (fa > fb) - (fa < fb);
}

float apply_median_filter(int sensor_id, float new_value) {
    static float buffers[5][MEDIAN_WINDOW_SIZE];
    static int indices[5] = {0};
    static int counts[5] = {0};
    
    // 更新缓冲区
    buffers[sensor_id][indices[sensor_id]] = new_value;
    indices[sensor_id] = (indices[sensor_id] + 1) % MEDIAN_WINDOW_SIZE;
    
    if (counts[sensor_id] < MEDIAN_WINDOW_SIZE) {
        counts[sensor_id]++;
    }
    
    // 复制并排序数据
    float temp[MEDIAN_WINDOW_SIZE];
    memcpy(temp, buffers[sensor_id], counts[sensor_id] * sizeof(float));
    qsort(temp, counts[sensor_id], sizeof(float), compare_float);
    
    // 返回中值
    float median = temp[counts[sensor_id] / 2];
    
    ESP_LOGV(TAG, "Sensor %d median: %.3f", sensor_id, median);
    return median;
}
```

**说明：**
- 均值滤波去除高频噪声，平滑信号
- 中值滤波消除脉冲干扰，保持边缘特征
- 循环缓冲区实现高效的滑动窗口

## 2. 姿态计算与传感器融合

### 2.1 四元数运算

**四元数归一化公式：**
```
q_normalized = q / |q|
|q| = √(w² + x² + y² + z²)
```

**代码实现：**
```c
// mpu6050_driver.c - 四元数运算
typedef struct {
    float w, x, y, z;
} quaternion_t;

void quaternion_normalize(quaternion_t* q) {
    // 计算四元数模长
    float norm = sqrtf(q->w * q->w + q->x * q->x + 
                      q->y * q->y + q->z * q->z);
    
    if (norm > 0.0001f) {  // 避免除零
        float inv_norm = 1.0f / norm;
        q->w *= inv_norm;
        q->x *= inv_norm;
        q->y *= inv_norm;
        q->z *= inv_norm;
    }
    
    ESP_LOGV(TAG, "Quaternion normalized: w=%.3f, x=%.3f, y=%.3f, z=%.3f",
             q->w, q->x, q->y, q->z);
}

// 四元数乘法（旋转合成）
quaternion_t quaternion_multiply(const quaternion_t* q1, const quaternion_t* q2) {
    quaternion_t result;
    
    result.w = q1->w * q2->w - q1->x * q2->x - q1->y * q2->y - q1->z * q2->z;
    result.x = q1->w * q2->x + q1->x * q2->w + q1->y * q2->z - q1->z * q2->y;
    result.y = q1->w * q2->y - q1->x * q2->z + q1->y * q2->w + q1->z * q2->x;
    result.z = q1->w * q2->z + q1->x * q2->y - q1->y * q2->x + q1->z * q2->w;
    
    return result;
}
```

### 2.2 互补滤波器

**互补滤波公式：**
```
θ = α * θ_gyro + (1-α) * θ_accel
```

**代码实现：**
```c
// 互补滤波器实现
typedef struct {
    float roll, pitch, yaw;
} euler_angles_t;

euler_angles_t complementary_filter(float ax, float ay, float az,
                                   float gx, float gy, float gz,
                                   float dt) {
    static euler_angles_t angles = {0};
    const float ALPHA = 0.98f;  // 互补滤波系数
    
    // 陀螺仪角度积分
    float gyro_roll = angles.roll + gx * dt;
    float gyro_pitch = angles.pitch + gy * dt;
    float gyro_yaw = angles.yaw + gz * dt;
    
    // 加速度计角度计算
    float accel_roll = atan2f(ay, az) * 180.0f / M_PI;
    float accel_pitch = atan2f(-ax, sqrtf(ay*ay + az*az)) * 180.0f / M_PI;
    
    // 互补滤波融合
    angles.roll = ALPHA * gyro_roll + (1.0f - ALPHA) * accel_roll;
    angles.pitch = ALPHA * gyro_pitch + (1.0f - ALPHA) * accel_pitch;
    angles.yaw = gyro_yaw;  // 偏航角只能用陀螺仪
    
    ESP_LOGD(TAG, "Angles: Roll=%.2f°, Pitch=%.2f°, Yaw=%.2f°",
             angles.roll, angles.pitch, angles.yaw);
    
    return angles;
}
```

**说明：**
- α值接近1时更信任陀螺仪（短期精确）
- α值接近0时更信任加速度计（长期稳定）
- 典型值α=0.98平衡了精度和稳定性

### 2.3 卡尔曼滤波器

**卡尔曼滤波方程组：**
```
预测: x̂⁻ = F*x̂ + B*u
     P⁻ = F*P*Fᵀ + Q

更新: K = P⁻*Hᵀ*(H*P⁻*Hᵀ + R)⁻¹
     x̂ = x̂⁻ + K*(z - H*x̂⁻)
     P = (I - K*H)*P⁻
```

**代码实现：**
```c
// 简化的1D卡尔曼滤波器
typedef struct {
    float x;        // 状态估计
    float P;        // 估计协方差
    float Q;        // 过程噪声
    float R;        // 测量噪声
} kalman_filter_t;

void kalman_init(kalman_filter_t* kf, float Q, float R) {
    kf->x = 0.0f;
    kf->P = 1.0f;
    kf->Q = Q;
    kf->R = R;
}

float kalman_update(kalman_filter_t* kf, float measurement) {
    // 预测步骤
    // x̂⁻ = x̂ (假设F=1, B=0)
    // P⁻ = P + Q
    kf->P += kf->Q;
    
    // 更新步骤
    // K = P⁻ / (P⁻ + R)
    float K = kf->P / (kf->P + kf->R);
    
    // x̂ = x̂⁻ + K*(z - x̂⁻)
    kf->x = kf->x + K * (measurement - kf->x);
    
    // P = (1 - K)*P⁻
    kf->P = (1.0f - K) * kf->P;
    
    ESP_LOGV(TAG, "Kalman: measurement=%.3f, estimate=%.3f, K=%.3f",
             measurement, kf->x, K);
    
    return kf->x;
}

// 使用示例
static kalman_filter_t flex_kalman_filters[5];

void init_kalman_filters(void) {
    for (int i = 0; i < 5; i++) {
        kalman_init(&flex_kalman_filters[i], 0.01f, 0.1f);
    }
}
```

## 3. 系统性能优化

### 3.1 功耗计算

**功耗公式：**
```
P_total = P_static + P_dynamic
P_dynamic = α * C * V² * f
```

**代码实现：**
```c
// main.c - 功耗监控
typedef struct {
    float voltage;      // 工作电压
    float current;      // 工作电流
    float power;        // 功耗
    uint32_t timestamp; // 时间戳
} power_measurement_t;

void monitor_power_consumption(void) {
    static uint32_t last_measurement = 0;
    uint32_t current_time = esp_timer_get_time() / 1000; // ms
    
    if (current_time - last_measurement > 1000) { // 每秒测量一次
        // 估算当前功耗（实际应用中需要外部功耗测量电路）
        float estimated_current = 0.0f;
        
        // CPU功耗估算
        uint32_t cpu_freq = esp_clk_cpu_freq() / 1000000; // MHz
        estimated_current += cpu_freq * 0.001f; // 经验公式：1mA/MHz
        
        // 传感器功耗
        estimated_current += 5 * 0.5f; // 5个传感器，每个0.5mA
        estimated_current += 3.5f;     // MPU6050功耗
        
        // WiFi/蓝牙功耗（如果启用）
        #ifdef CONFIG_WIFI_ENABLED
        estimated_current += 80.0f;    // WiFi活跃时功耗
        #endif
        
        float voltage = 3.3f;
        float power = voltage * estimated_current / 1000.0f; // W
        
        ESP_LOGI(TAG, "Power consumption: %.1fmA, %.3fW @ %.1fV", 
                 estimated_current, power, voltage);
        
        last_measurement = current_time;
    }
}
```

### 3.2 实时性能分析

**响应时间计算：**
```
T_total = T_sensor + T_processing + T_transmission
```

**代码实现：**
```c
// 性能监控结构体
typedef struct {
    uint64_t sensor_read_time;
    uint64_t processing_time;
    uint64_t transmission_time;
    uint64_t total_time;
    uint32_t sample_count;
} performance_stats_t;

static performance_stats_t perf_stats = {0};

void measure_performance(void) {
    uint64_t start_time, end_time;
    
    // 测量传感器读取时间
    start_time = esp_timer_get_time();
    sensor_data_t sensor_data;
    read_all_sensors(&sensor_data);
    end_time = esp_timer_get_time();
    perf_stats.sensor_read_time += (end_time - start_time);
    
    // 测量数据处理时间
    start_time = esp_timer_get_time();
    process_sensor_data(&sensor_data);
    end_time = esp_timer_get_time();
    perf_stats.processing_time += (end_time - start_time);
    
    // 测量传输时间
    start_time = esp_timer_get_time();
    transmit_data(&sensor_data);
    end_time = esp_timer_get_time();
    perf_stats.transmission_time += (end_time - start_time);
    
    perf_stats.sample_count++;
    
    // 每100个样本输出一次统计
    if (perf_stats.sample_count % 100 == 0) {
        uint32_t count = perf_stats.sample_count;
        
        ESP_LOGI(TAG, "Performance Stats (avg over %lu samples):", count);
        ESP_LOGI(TAG, "  Sensor read: %lld μs", 
                 perf_stats.sensor_read_time / count);
        ESP_LOGI(TAG, "  Processing:  %lld μs", 
                 perf_stats.processing_time / count);
        ESP_LOGI(TAG, "  Transmission: %lld μs", 
                 perf_stats.transmission_time / count);
        ESP_LOGI(TAG, "  Total latency: %lld μs", 
                 (perf_stats.sensor_read_time + 
                  perf_stats.processing_time + 
                  perf_stats.transmission_time) / count);
    }
}
```

### 3.3 自适应采样率

**采样率优化公式：**
```
f_optimal = f_max * α + f_min * (1-α)
α = activity_level / max_activity
```

**代码实现：**
```c
// 自适应采样率控制
typedef struct {
    float current_rate;     // 当前采样率
    float min_rate;        // 最小采样率
    float max_rate;        // 最大采样率
    float activity_level;  // 活动水平
} adaptive_sampling_t;

static adaptive_sampling_t sampling_ctrl = {
    .current_rate = 100.0f,
    .min_rate = 10.0f,
    .max_rate = 200.0f,
    .activity_level = 0.0f
};

float calculate_activity_level(const sensor_data_t* current, 
                              const sensor_data_t* previous) {
    float activity = 0.0f;
    
    // 计算传感器变化幅度
    for (int i = 0; i < 5; i++) {
        float diff = fabsf(current->flex_values[i] - previous->flex_values[i]);
        activity += diff;
    }
    
    // 计算IMU变化幅度
    float accel_diff = sqrtf(
        powf(current->mpu_accel[0] - previous->mpu_accel[0], 2) +
        powf(current->mpu_accel[1] - previous->mpu_accel[1], 2) +
        powf(current->mpu_accel[2] - previous->mpu_accel[2], 2)
    );
    activity += accel_diff * 100.0f; // 放大IMU贡献
    
    // 归一化到0-1范围
    const float MAX_EXPECTED_ACTIVITY = 5000.0f;
    activity = fminf(activity / MAX_EXPECTED_ACTIVITY, 1.0f);
    
    return activity;
}

void update_sampling_rate(const sensor_data_t* current, 
                         const sensor_data_t* previous) {
    // 计算当前活动水平
    float activity = calculate_activity_level(current, previous);
    
    // 滑动平均滤波
    const float ALPHA = 0.1f;
    sampling_ctrl.activity_level = ALPHA * activity + 
                                  (1.0f - ALPHA) * sampling_ctrl.activity_level;
    
    // 计算新的采样率
    float rate_range = sampling_ctrl.max_rate - sampling_ctrl.min_rate;
    sampling_ctrl.current_rate = sampling_ctrl.min_rate + 
                                rate_range * sampling_ctrl.activity_level;
    
    // 更新任务延迟
    uint32_t delay_ms = (uint32_t)(1000.0f / sampling_ctrl.current_rate);
    
    ESP_LOGD(TAG, "Activity: %.3f, Sampling rate: %.1f Hz, Delay: %lu ms",
             sampling_ctrl.activity_level, sampling_ctrl.current_rate, delay_ms);
    
    // 返回延迟时间供任务使用
    vTaskDelay(pdMS_TO_TICKS(delay_ms));
}
```

## 4. 统计分析与质量评估

### 4.1 信号质量评估

**信噪比计算：**
```
SNR = 10 * log10(P_signal / P_noise)
```

**代码实现：**
```c
// 信号质量分析
typedef struct {
    float mean;
    float variance;
    float std_dev;
    float snr_db;
    uint32_t sample_count;
} signal_quality_t;

void analyze_signal_quality(float* samples, int count, signal_quality_t* quality) {
    // 计算均值
    float sum = 0.0f;
    for (int i = 0; i < count; i++) {
        sum += samples[i];
    }
    quality->mean = sum / count;
    
    // 计算方差
    float var_sum = 0.0f;
    for (int i = 0; i < count; i++) {
        float diff = samples[i] - quality->mean;
        var_sum += diff * diff;
    }
    quality->variance = var_sum / (count - 1);
    quality->std_dev = sqrtf(quality->variance);
    
    // 估算信噪比（简化计算）
    float signal_power = quality->mean * quality->mean;
    float noise_power = quality->variance;
    
    if (noise_power > 0.001f) {
        quality->snr_db = 10.0f * log10f(signal_power / noise_power);
    } else {
        quality->snr_db = 60.0f; // 假设很高的SNR
    }
    
    quality->sample_count = count;
    
    ESP_LOGI(TAG, "Signal Quality - Mean: %.3f, StdDev: %.3f, SNR: %.1fdB",
             quality->mean, quality->std_dev, quality->snr_db);
}
```

### 4.2 系统可靠性监控

**MTBF计算：**
```
MTBF = Total_Operating_Time / Number_of_Failures
```

**代码实现：**
```c
// 可靠性监控
typedef struct {
    uint64_t start_time;
    uint64_t total_runtime;
    uint32_t error_count;
    uint32_t reset_count;
    float mtbf_hours;
} reliability_stats_t;

static reliability_stats_t reliability = {0};

void init_reliability_monitor(void) {
    reliability.start_time = esp_timer_get_time();
    
    // 检查重启原因
    esp_reset_reason_t reset_reason = esp_reset_reason();
    if (reset_reason != ESP_RST_POWERON) {
        reliability.reset_count++;
        ESP_LOGW(TAG, "System reset detected, reason: %d", reset_reason);
    }
}

void update_reliability_stats(bool error_occurred) {
    uint64_t current_time = esp_timer_get_time();
    reliability.total_runtime = current_time - reliability.start_time;
    
    if (error_occurred) {
        reliability.error_count++;
        ESP_LOGW(TAG, "Error detected, total errors: %lu", reliability.error_count);
    }
    
    // 计算MTBF（小时）
    if (reliability.error_count > 0) {
        reliability.mtbf_hours = (reliability.total_runtime / 1000000.0f / 3600.0f) / 
                                reliability.error_count;
    } else {
        reliability.mtbf_hours = reliability.total_runtime / 1000000.0f / 3600.0f;
    }
    
    // 每小时输出一次统计
    static uint64_t last_report = 0;
    if (current_time - last_report > 3600000000ULL) { // 1小时
        ESP_LOGI(TAG, "Reliability Stats:");
        ESP_LOGI(TAG, "  Runtime: %.2f hours", 
                 reliability.total_runtime / 1000000.0f / 3600.0f);
        ESP_LOGI(TAG, "  Errors: %lu", reliability.error_count);
        ESP_LOGI(TAG, "  Resets: %lu", reliability.reset_count);
        ESP_LOGI(TAG, "  MTBF: %.2f hours", reliability.mtbf_hours);
        
        last_report = current_time;
    }
}
```

## 5. 使用示例

### 在main.c中的完整集成示例：

```c
void app_main(void) {
    // 初始化所有组件
    init_reliability_monitor();
    init_kalman_filters();
    flex_sensors_init();
    mpu6050_init();
    
    sensor_data_t current_data, previous_data = {0};
    
    while (1) {
        uint64_t loop_start = esp_timer_get_time();
        
        // 读取传感器数据
        read_all_sensors(&current_data);
        
        // 应用滤波算法
        for (int i = 0; i < 5; i++) {
            float voltage = flex_adc_to_voltage(current_data.flex_values[i]);
            current_data.flex_values[i] = (int)apply_mean_filter(i, voltage * 1000);
        }
        
        // 姿态计算
        euler_angles_t angles = complementary_filter(
            current_data.mpu_accel[0], current_data.mpu_accel[1], current_data.mpu_accel[2],
            current_data.mpu_gyro[0], current_data.mpu_gyro[1], current_data.mpu_gyro[2],
            0.01f // 10ms采样周期
        );
        
        // 传输数据
        transmit_sensor_data(&current_data);
        
        // 性能监控
        measure_performance();
        monitor_power_consumption();
        update_reliability_stats(false);
        
        // 自适应采样率
        update_sampling_rate(&current_data, &previous_data);
        previous_data = current_data;
        
        uint64_t loop_time = esp_timer_get_time() - loop_start;
        ESP_LOGV(TAG, "Loop time: %lld μs", loop_time);
    }
}
```
